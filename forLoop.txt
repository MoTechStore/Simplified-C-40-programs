#include <iostream>
using namespace std;

int main()
{
	// C++ program to print elements of array using for loop 
	string myarray[20]= {"Robotics","C++","Python","AI","ML"};
	cout<<"Elements Of myarray :"<<endl;
	for (int i=0; i<4; i++)
	{
		cout<<myarray[i]<<endl;	
	}
   
	return 0;
}
