#include <iostream>
using namespace std;

int main()
{
	// C++ program to check if number is even or odd
	
	int number;
	cout<<"Enter Any Number : ";
	cin>>number;
	
	if ( number%2==0)
       cout<<"Number "<<number<<" Is Even Number ";
	 else
	   cout<<"Number "<<number<<" Is Odd ";	

	return 0;
}
